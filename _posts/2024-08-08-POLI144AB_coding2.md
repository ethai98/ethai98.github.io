vdem_global_mean <- vdem_cleaned %>%
  group_by(year) %>%
  summarise(v2x_polyarchy = mean(v2x_polyarchy, na.rm = TRUE)) %>%
  mutate(country_name = "Global")

vdem_cleaned %>%
  filter(country_name == "United States of America") %>%
  select(-v2mecenefm) %>%
  bind_rows(vdem_global_mean) %>%
